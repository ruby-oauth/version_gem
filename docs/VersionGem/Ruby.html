def actual_minor_version?(major, minor, engine = "ruby")
  segs = RUBY_VER.segments
  major.to_i == segs[0] &&
    minor.to_i == segs[1] &&
    ::RUBY_ENGINE == engine
end

def gte_minimum_version?(version, engine = "ruby")
  RUBY_VER >= ::Gem::Version.new(version) && ::RUBY_ENGINE == engine
end
